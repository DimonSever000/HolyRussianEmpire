
# Arkhangelsk republic
arkhangelsk_republic = {
	icon = "merchant"
	allow_normal_conversion = yes
	potential = {
		NOT = { has_reform = great_veche_republic }
		NOT = { have_had_reform = great_veche_republic }
		NOT = { has_reform = veche_republic }
		NOT = { have_had_reform = veche_republic }
		OR = { tag = R1A }
	}
	legacy_equivalent = arkhangelsk_republic_legacy
	modifiers = {
		build_cost = -0.1
		embracement_cost = -0.1
		max_absolutism = -20

	valid_for_nation_designer = no

	fixed_rank = 1

		enables_plutocratic_idea_group = yes

		elections_increase_factions_influence = yes

		trade_city_reform = trading_city
		has_parliament = yes
		government_abilities = {
			russian_mechanic
		}
	}
	duration = 4
}
